{{- if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "python-info-service.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if eq .Values.service.type "LoadBalancer" }}
  NOTE: It may take a few minutes for the LoadBalancer external IP to become available.
  kubectl get svc --namespace {{ .Release.Namespace }} {{ include "python-info-service.fullname" . }} -w
{{- else }}
  kubectl port-forward --namespace {{ .Release.Namespace }} service/{{ include "python-info-service.fullname" . }} 8080:{{ .Values.service.port }}
{{- end }}
